```python
import torch
from brain_ai import HippocampalSimulator, NeocortexArchitecture

class SmartMemorySystem:
    def __init__(self):
        self.hippocampus = HippocampalSimulator()
        self.neocortex = NeocortexArchitecture()
        
    def learn(self, data, labels):
        """å­¦ä¹ æ–°ä¿¡æ¯"""
        # å¿«é€Ÿç¼–ç åˆ°æµ·é©¬ä½“
        encoding = self.hippocampus.encode_memory(data)
        
        # å±‚æ¬¡åŒ–æŠ½è±¡åˆ°æ–°çš®å±‚
        abstraction = self.neocortex.process({
            "input": encoding['final_encoding'],
            "type": "learning"
        })
        
        return {
            "memory_id": self.hippocampus.store_memory(
                encoding['final_encoding'],
                metadata={"labels": labels.tolist()}
            ),
            "abstraction": abstraction
        }
    
    def recall(self, query):
        """æ£€ç´¢è®°å¿†"""
        query_encoding = self.hippocampus.encode_memory(query)
        return self.hippocampus.retrieve_memory(query_encoding['final_encoding'])

    result = system.learn(sample_data, sample_labels)

    # æ£€ç´¢ç¤ºä¾‹
```

```python
# continual_classifier.py
import torch
import torch.nn as nn
from brain_ai import ElasticWeightConsolidation, GenerativeReplay

class ContinualClassifier:
    def __init__(self, input_size=784, hidden_size=512, num_classes=10):
        self.network = nn.Sequential(
            nn.Linear(input_size, hidden_size),
            nn.ReLU(),
            nn.Linear(hidden_size, hidden_size),
            nn.ReLU(),
            nn.Linear(hidden_size, num_classes)
        )
        
        self.ewc = ElasticWeightConsolidation()
        self.replay = GenerativeReplay()
        
    def learn_task(self, task_id, train_loader):
        """å­¦ä¹ æ–°ä»»åŠ¡"""
        # ä½¿ç”¨EWCé˜²æ­¢ç¾éš¾æ€§é—å¿˜
        for epoch in range(10):
            for batch_x, batch_y in train_loader:
                self.ewc.learn_task(task_id, batch_x, batch_y, self.network)
        
        # ç”Ÿæˆå›æ”¾æ•°æ®
        self.replay.generate_replay_data(task_id, train_loader)
    
    def evaluate_all_tasks(self, test_loaders):
        """è¯„ä¼°æ‰€æœ‰ä»»åŠ¡"""
        results = []
        for task_id, test_loader in enumerate(test_loaders):
            accuracy = self.evaluate_task(task_id, test_loader)
            results.append(accuracy)
            print(f"Task {task_id}: {accuracy:.3f}")
        
        return results
    
    def evaluate_task(self, task_id, test_loader):
        """è¯„ä¼°å•ä¸ªä»»åŠ¡"""
        correct, total = 0, 0
        
        with torch.no_grad():
            for batch_x, batch_y in test_loader:
                outputs = self.network(batch_x)
                _, predicted = torch.max(outputs.data, 1)
                total += batch_y.size(0)
                correct += (predicted == batch_y).sum().item()
        
        return correct / total

# ä½¿ç”¨ç¤ºä¾‹

    # æ¨¡æ‹Ÿå¤šä»»åŠ¡å­¦ä¹

        # ç”Ÿæˆä»»åŠ¡æ•°æ®

        train_labels = torch.randint(0, 10, (1000,))
        train_loader = [(train_data, train_labels)]
        
        classifier.learn_task(task_id, train_loader)
        print(f"Task {task_id} å­¦ä¹ å®Œæˆ")
```
